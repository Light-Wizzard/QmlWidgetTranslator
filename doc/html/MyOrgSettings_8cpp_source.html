516QString MyOrgSettings::getFileInfo(MyOrgSettings::MyFileinfo thisInfo, const QString &thisFileFolder)
517{
518 setMessage("get File Info", QString("getFileInfo(%1)").arg(thisFileFolder), Debug);
519 if (thisFileFolder.isEmpty()) { return ""; }
520 //
521 QFileInfo theFileInfo(thisFileFolder);
522 //
523 switch (thisInfo)
524 {
525 case AbsoluteFilePath:
526 // The absolute path name consists of the full path and the file name.
527 return theFileInfo.absoluteFilePath();
528 case AbsolutePath:
529 // Returns a file's path absolute path. This doesn't include the file name..
530 return theFileInfo.absolutePath();
531 case BaseName:
532 // Returns the base name of the file without the path.
533 return theFileInfo.baseName();
534 case BirthTime:
535 // Returns the date and time when the file was created / born.
536 return theFileInfo.birthTime().toString();
537 case CanonicalPath:
538 // Returns the file's path canonical path (excluding the file name), i.e. an absolute path without symbolic links or redundant "." or ".." elements.
539 return theFileInfo.canonicalPath();
540 case CanonicalFilePath:
541 // Returns the canonical path including the file name, i.e. an absolute path without symbolic links or redundant "." or ".." elements.
542 return theFileInfo.canonicalFilePath();
543 case CompleteBaseName:
544 // Returns the complete base name of the file without the path.
545 // "/tmp/archive.tar.gz" == "archive.tar"
546 return theFileInfo.completeBaseName();
547 case CompleteSuffix:
548 // Returns the complete suffix (extension) of the file.
549 // "/tmp/archive.tar.gz" == "tar.gz"
550 return theFileInfo.completeSuffix();
551 case Directory:
552 // Returns the path of the object's parent directory as a QDir object.
553 return theFileInfo.dir().currentPath();
554 case FileName:
555 // Returns the name of the file, excluding the path. getFileInfo(FileName, "")
556 return theFileInfo.fileName();
557 case FilePath:
558 // Returns the file name, including the path (which may be absolute or relative).
559 return theFileInfo.filePath();
560 case IsWritable:
561 // Returns true if the user can write to the file; otherwise returns false.
562 if (theFileInfo.isWritable()) return "true"; else return "false";
563 case FileSize:
564 return QString::number(theFileInfo.size());
565 case IsFile:
566 if (isFileExists(thisFileFolder)) return "true"; else return "false";
567 case IsFolder:
568 if (isPathExists(thisFileFolder)) return "true"; else return "false";
569 case IsSymLink:
570 if (theFileInfo.isSymLink()) return "true"; else return "false";
571 case SymLinkTarget:
572 if (theFileInfo.isSymLink()) return theFileInfo.symLinkTarget(); else return "";
573 }
574 return "";
575}
