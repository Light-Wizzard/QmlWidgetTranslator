212QString QGumboNode::outerHtml() const
213{
214 Q_ASSERT(ptr_);
215
216 QString text;
217 switch (ptr_->type) {
218 case GUMBO_NODE_DOCUMENT: {
219 throw std::runtime_error("invalid node type");
220 }
221 case GUMBO_NODE_ELEMENT: {
222 const auto& elem = ptr_->v.element;
223 const auto& tag = elem.original_tag;
224 if (tag.data && tag.length) {
225 int lenght = elem.end_pos.offset - elem.start_pos.offset + elem.original_end_tag.length;
226 Q_ASSERT(lenght > 0);
227 text = QString::fromUtf8(tag.data, lenght);
228 }
229 break;
230 }
231 default: {
232 const auto& str = ptr_->v.text.original_text;
233 text = QString::fromUtf8(str.data, str.length);
234 }}
235 return text;
236}
